# The FizzBuzz Game

```C
#include <stdio.h>

int main() {
    int i, n;
    printf("Enter n: ");
    scanf("%d", & n);

    if (n % 3 == 0 && n % 5 == 0) {
        printf("FizzBuzz");
    } else if (n % 3 == 0) {
        printf("Fizz");
    } else if (n % 5 == 0) {
        printf("Buzz");
    }
    return 0;
}
```
